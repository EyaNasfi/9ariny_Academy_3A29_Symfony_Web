package services;





import entities.salle;
import utils.MyDB;

import java.sql.*;
import java.util.ArrayList;
import java.util.List;

public class ServiceSalle implements ISalle<salle> {

    private Connection connection;

    public ServiceSalle() {

        connection = MyDB.getInstance().getConnection();
    }

    @Override
    public void ajouter(salle salle) throws SQLException {
        String req = "insert into salle (matiere,prof)" +"values('" + salle.getMatiere()+ "','" + salle.getProf() +  "')";
        Statement ste = connection.createStatement();
        ste.executeUpdate(req);
    }

    @Override
    public void modifier(salle salle) throws SQLException {
        String req = "update salle set matiere=?,prof=? where id=?";
        PreparedStatement pre = connection.prepareStatement(req);
        pre.setInt(1, salle.getId());
        pre.setString(2, salle.getMatiere());
        pre.setString(3, salle.getProf());


        pre.executeUpdate();
    }

    @Override
    public void supprimer(salle salle) throws SQLException {
        String req = "delete from salle where id=?";
        PreparedStatement pre = connection.prepareStatement(req);
        pre.setInt(1, salle.getId());
        pre.executeUpdate();
    }

    @Override
    public List<salle> afficher() throws SQLException {
        String req = "select * from salle";
        Statement ste = connection.createStatement();
        ResultSet res = ste.executeQuery(req);
        List<salle> list = new ArrayList<>();
        while (res.next()) {
            salle s = new salle();
            s.setId(res.getInt(1));
            s.setMatiere(res.getString("matiere"));
            s.setProf(res.getString("prof"));

            list.add(s);
        }

        return list;
    }


}

package utils;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class MyDB {
    private  final  String URL = "jdbc:mysql://localhost:3306/mydatabase";
    private  final  String USER = "root";
    private  final  String PWD = "";
    private Connection connection; // Single declaration

    private static MyDB instance;

    private MyDB() {
        try {
            connection = DriverManager.getConnection(URL, USER, PWD);
            System.out.println("Connected to DB");
        } catch (SQLException e) {
            System.out.println(e.getMessage());
        }
    }

    public static MyDB getInstance() {
        if (instance == null) {
            instance = new MyDB();
        }
        return instance;
    }

    public Connection getConnection() {
        return connection;
    }

}
